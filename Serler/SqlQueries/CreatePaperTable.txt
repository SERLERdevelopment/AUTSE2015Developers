CREATE TABLE dbo.Paper (PaperId int PRIMARY KEY IDENTITY(1,1),
			PaperTitle nvarchar(500) NOT NULL,
			Rating int,
			JournalName nvarchar(500),
			NoOfPeopleRated int,
			Author nvarchar(500) NOT NULL,
			Date nvarchar (10) NOT NULL,
			Publisher nvarchar(500),			
			Category nvarchar(500) NOT NULL,
			Methodology nvarchar (500),
			Practice nvarchar (500),
			OutcomeBeingTested nvarchar (500),
			ContextWho nvarchar (500),
			ContextWhat nvarchar (500),
			ContextWhere nvarchar (500),
			StudyResult nvarchar (500),
			ImplementationIntegrity nvarchar (500),
			ConfidenceRating int,
			ConfidenceRatingReason nvarchar (500),
			WhoRated nvarchar (500),
			ResearchQuestion nvarchar (500),
			ResearchMethod nvarchar (500),
			ResearchMetrics nvarchar (500),
			ResearchLevel nvarchar (500),
			ParticipantsNature nvarchar (500),
			isAnalyzed bit,
			IsActive bit,
			isRejected bit)
